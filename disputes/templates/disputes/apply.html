// Validate current step before moving forward function validateStep(step) { // Clear any existing validation errors for this step clearValidationErrors(); if (step === 1) { // Validate Step 1: Applicant Information var applicantName = document.getElementById('id_applicant_name'); var applicantSurname = document.getElementById('id_applicant_surname'); var applicantCell = document.getElementById('id_applicant_cell'); if (!applicantName.value.trim()) { showError('id_applicant_name', 'Please enter your first name'); return false; } if (!applicantSurname.value.trim()) { showError('id_applicant_surname', 'Please enter your last name'); return false; } if (!applicantCell.value.trim()) { showError('id_applicant_cell', 'Please enter your cell phone number'); return false; } if (!/^\d+$/.test(applicantCell.value.trim())) { showError('id_applicant_cell', 'Please enter numbers only for cell phone'); return false; } } else if (step === 2) { // Validate Step 2: Dispute Details var disputeType = document.querySelector('input[name="dispute_type"]:checked'); if (!disputeType) { showError('id_dispute_type', 'Please select a dispute type'); return false; } var description = document.getElementById('id_description'); if (!description || !description.value.trim()) { showError('id_description', 'Please describe your dispute in detail'); return false; } } else if (step === 3) { // Step 3: Documents (optional, no validation needed) return true; } else if (step === 4) { // Step 4: Respondent Type (radio buttons, always selected) return true; } else if (step === 5) { // Validate Step 5: Respondent Information var respondentType = document.querySelector('input[name="respondent_type"]:checked'); if (!respondentType) { showError('respondent_type', 'Please select respondent type'); return false; } if (respondentType.value === 'ind') { // Individual respondent validation var respondentName = document.getElementById('ind_respondent_name'); var respondentSurname = document.getElementById('id_respondent_surname'); var respondentCell = document.getElementById('id_respondent_cell'); if (!respondentName || !respondentName.value.trim()) { showError('ind_respondent_name', 'Please enter respondent\'s first name'); return false; } if (!respondentSurname || !respondentSurname.value.trim()) { showError('id_respondent_surname', 'Please enter respondent\'s last name'); return false; } if (!respondentCell || !respondentCell.value.trim()) { showError('id_respondent_cell', 'Please enter respondent\'s cell phone number'); return false; } if (!/^\d+$/.test(respondentCell.value.trim())) { showError('id_respondent_cell', 'Please enter numbers only for cell phone'); return false; } } else if (respondentType.value === 'bus') { // Business respondent validation var businessName = document.getElementById('bus_business_name'); var ownerName = document.getElementById('id_owner_name'); if (!businessName || !businessName.value.trim()) { showError('bus_business_name', 'Please enter business name'); return false; } if (!ownerName || !ownerName.value.trim()) { showError('id_owner_name', 'Please enter owner/representative first name'); return false; } } } return true; } function nextStep() { // Only proceed if current step is valid if (validateStep(currentStep)) { if (currentStep < totalSteps) { currentStep++; updateStepDisplay(); window.scrollTo({top: 0, behavior: 'smooth'}); } } } function showError(fieldId, message) { // Remove existing error messages clearValidationErrors(); // Focus on the field with error var field = document.getElementById(fieldId); if (field) { field.focus(); field.classList.add('is-invalid'); // Create error message var errorDiv = document.createElement('div'); errorDiv.className = 'invalid-feedback'; errorDiv.style.display = 'block'; errorDiv.textContent = message; // Add error message after the field if (field.parentElement) { field.parentElement.appendChild(errorDiv); } // Scroll to the error field field.scrollIntoView({behavior: 'smooth', block: 'center'}); } } function clearValidationErrors() { // Remove all error messages var invalidFeedbacks = document.querySelectorAll('.invalid-feedback'); invalidFeedbacks.forEach(function(el) { el.remove(); }); // Remove invalid classes from form controls var invalidFields = document.querySelectorAll('.is-invalid'); invalidFields.forEach(function(el) { el.classList.remove('is-invalid'); }); }
